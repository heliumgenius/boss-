```python
"""Rule-based natural language → structured search parameters."""

from __future__ import annotations

import re
from typing import Any

from ..constants import CITY_CODES, DEGREE_CODES, EXP_CODES, SALARY_CODES

def _match_city(text: str) -> tuple[str | None, str]:
    """Try to match a known city name; return (city_name, remaining_text)."""
    # Sort by length descending so "哈尔滨" matches before "哈"
    for city in sorted(CITY_CODES, key=len, reverse=True):
        if city in text:
            idx = text.index(city)
            remaining = text[:idx] + text[idx + len(city):]
            return city, remaining.strip()
    return None, text


def _match_salary(text: str) -> tuple[str | None, str]:
    """Match salary patterns like 20k以上, 20-30k, 15K-25K, 30K以下."""
    # "30k以上" or "30K以上"
    m = re.search(r'(\d+)\s*[kK]\s*以上', text)
    if m:
        val = int(m.group(1))
        remaining = text[:m.start()] + text[m.end():]
        for label, code in SALARY_CODES.items():
            if label.startswith(f"{val}-") or (label == f"{val}K以上".replace("K", "K")):
                return label, remaining.strip()
        # Fallback: find the smallest bracket whose lower bound >= val
        best = None
        for label in SALARY_CODES:
            if label.endswith("K以上") or label.endswith("K以下"):
                continue
            parts = label.replace("K", "").split("-")
            if len(parts) == 2 and parts[0].isdigit():
                low = int(parts[0])
                # If val is between low and low+10, or val == low
                if low <= val < low + 10:
                    best = label
        return best or f"{val}-{val+10}K", remaining.strip()

    # "20k-30k" or "15K-25K"
    m = re.search(r'(\d+)\s*[kK]\s*[-~到]\s*(\d+)\s*[kK]', text)
    if m:
        low, high = int(m.group(1)), int(m.group(2))
        remaining = text[:m.start()] + text[m.end():]
        for label in SALARY_CODES:
            if label.startswith(f"{low}-") and label.endswith(f"{high}K"):
                return label, remaining.strip()
        return f"{low}-{high}K", remaining.strip()

    # "30k以下"
    m = re.search(r'(\d+)\s*[kK]\s*以下', text)
    if m:
        val = int(m.group(1))
        remaining = text[:m.start()] + text[m.end():]
        for label in SALARY_CODES:
            if label == f"{val}K以下".replace("K", "K"):
                return label, remaining.strip()
        return None, remaining.strip()

    return None, text


def _match_experience(text: str) -> tuple[str | None, str]:
    """Match experience like 3年, 3-5年, 3年以上, 1年以内."""
    m = re.search(r'(\d+)\s*[-~到]\s*(\d+)\s*年', text)
    if m:
        low = int(m.group(1))
        remaining = text[:m.start()] + text[m.end():]
        for label in EXP_CODES:
            if label == f"{low}-{m.group(2)}年":
                return label, remaining.strip()
        return None, remaining.strip()

    m = re.search(r'(\d+)\s*年\s*以上', text)
    if m:
        val = int(m.group(1))
        remaining = text[:m.start()] + text[m.end():]
        for label in EXP_CODES:
            if label.startswith(f"{val}-") or label == f"{val}年以上":
                return label, remaining.strip()
        return None, remaining.strip()

    m = re.search(r'(\d+)\s*年\s*以内', text)
    if m:
        val = int(m.group(1))
        remaining = text[:m.start()] + text[m.end():]
        for label in EXP_CODES:
            if label == f"{val}年以内":
                return label, remaining.strip()
        return None, remaining.strip()

    m = re.search(r'(\d+)\s*年', text)
    if m:
        val = int(m.group(1))
        remaining = text[:m.start()] + text[m.end():]
        for label in EXP_CODES:
            if label.startswith(f"{val}-"):
                return label, remaining.strip()
        return None, remaining.strip()

    return None, text


def _match_degree(text: str) -> tuple[str | None, str]:
    """Match degree like 本科, 硕士, 大专."""
    # Sort by length descending so exact match hits before substring match
    degrees = sorted(DEGREE_CODES, key=len, reverse=True)
    for deg in degrees:
        if deg == "不限":
            continue
        if deg in text:
            idx = text.index(deg)
            remaining = text[:idx] + text[idx + len(deg):]
            return deg, remaining.strip()
    return None, text


def parse_query(query: str) -> dict[str, Any]:
    """Parse a natural language search query into structured parameters.

    Returns a dict with keys: keyword, city, salary, experience, degree.
    Any key may be None if not found.
    """
    text = query.strip()

    city, text = _match_city(text)
    salary, text = _match_salary(text)
    experience, text = _match_experience(text)
    degree, text = _match_degree(text)

    # Remaining text is the keyword
    keyword = text.strip() or None

    return {
        "keyword": keyword,
        "city": city,
        "salary": salary,
        "experience": experience,
        "degree": degree,
    }
```

```python
"""FastAPI web application for boss-cli Web UI."""

from __future__ import annotations

import json
import time
from typing import Any, AsyncGenerator

from fastapi import FastAPI, Form, Request
from fastapi.responses import HTMLResponse, StreamingResponse
from fastapi.staticfiles import StaticFiles
from fastapi.templating import Jinja2Templates

from ..auth import load_credential
from ..client.transport import BossClient
from ..exceptions import BossApiError
from .parser import parse_query


def create_app() -> FastAPI:
    app = FastAPI(title="Boss CLI Web UI")

    templates = Jinja2Templates(directory=__file__.rsplit("/", 1)[0] + "/templates")

    @app.get("/", response_class=HTMLResponse)
    async def index(request: Request):
        return templates.TemplateResponse("index.html", {"request": request})

    @app.post("/search", response_class=HTMLResponse)
    async def search(query: str = Form(...)):
        params = parse_query(query)
        cred = load_credential()
        if not cred:
            return HTMLResponse('<div class="error">未登录，请先在终端执行 boss login</div>')

        city_code = None
        if params["city"]:
            from ..constants import CITY_CODES
            city_code = CITY_CODES.get(params["city"])

        salary_code = None
        if params["salary"]:
            from ..constants import SALARY_CODES
            salary_code = SALARY_CODES.get(params["salary"])

        exp_code = None
        if params["experience"]:
            from ..constants import EXP_CODES
            exp_code = EXP_CODES.get(params["experience"])

        degree_code = None
        if params["degree"]:
            from ..constants import DEGREE_CODES
            degree_code = DEGREE_CODES.get(params["degree"])

        try:
            with BossClient(cred) as client:
                data = client.search_jobs(
                    query=params["keyword"] or "",
                    city=city_code or "100010000",
                    experience=exp_code,
                    degree=degree_code,
                    salary=salary_code,
                )
        except BossApiError as e:
            return HTMLResponse(f'<div class="error">搜索失败: {e}</div>')

        job_list = data.get("jobList", [])
        lid = data.get("lid", "")

        rows = ""
        for i, job in enumerate(job_list, 1):
            name = job.get("jobName", "-")
            company = job.get("brandName", "-")
            salary_text = job.get("salaryDesc", "-")
            exp_text = job.get("jobExperience", "-")
            degree_text = job.get("jobDegree", "-")
            sec_id = job.get("securityId", "")
            rows += f"""<tr>
                <td><input type="checkbox" class="job-check" value="{sec_id}" data-lid="{lid}"></td>
                <td>{i}</td>
                <td>{name}</td>
                <td>{company}</td>
                <td>{salary_text}</td>
                <td>{exp_text}</td>
                <td>{degree_text}</td>
            </tr>"""

        return HTMLResponse(rows)

    return app


app = create_app()
```

```python
    @app.post("/batch-greet")
    async def batch_greet(data: dict):
        ids = data.get("ids", [])
        lid = data.get("lid", "")

        cred = load_credential()
        if not cred:
            return {"error": "not authenticated"}

        async def event_stream():
            yield f"event: start\ndata: {json.dumps({'total': len(ids)})}\n\n"
            success = 0
            fail = 0
            for i, sec_id in enumerate(ids):
                try:
                    with BossClient(cred) as client:
                        client.add_friend(security_id=sec_id, lid=lid)
                    success += 1
                    yield f"event: progress\ndata: {json.dumps({'index': i+1, 'security_id': sec_id, 'status': 'ok'})}\n\n"
                except BossApiError as e:
                    fail += 1
                    yield f"event: progress\ndata: {json.dumps({'index': i+1, 'security_id': sec_id, 'status': 'fail', 'message': str(e)})}\n\n"
                time.sleep(1.5)
            yield f"event: complete\ndata: {json.dumps({'success': success, 'fail': fail})}\n\n"

        return StreamingResponse(event_stream(), media_type="text/event-stream")
```
